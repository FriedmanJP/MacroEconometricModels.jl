```julia
function _aiyagari_clearing(price_fn::Function)
    return function (r_mid::T, params::Dict{Symbol,T}) where {T<:AbstractFloat}
        alpha = get(params, :alpha, T(0.36)); delta = get(params, :delta, T(0.025))
        Z = get(params, :Z, one(T)); L = get(params, :L, one(T))
        r_eff = r_mid + delta
        r_eff <= zero(T) && return (T(Inf), Dict{Symbol,T}(:r => r_mid))
        K_d = (r_eff / (alpha * Z * L^(one(T) - alpha)))^(one(T) / (alpha - one(T)))
        prices = price_fn(K_d, params); prices[:r] = r_mid
        return (K_d, prices)
    end
end
```

function compute_steady_state(spec::HADSGESpec{T};
                          K_init::T=T(10),
                          r_bounds::Union{Nothing,Tuple{T,T}}=nothing,
                          max_iter::Int=200, tol::Real=T(1e-8), verbose::Bool=false,
                          price_fn::Union{Nothing,Function}=nothing,
                          clearing::Union{Nothing,Function}=nothing) where {T<:AbstractFloat}
    pfn = isnothing(price_fn) ? _default_cobb_douglas_price_fn : price_fn
    params = copy(spec.het_params)
    for (k,v) in (:alpha=>T(0.36), :delta=>T(0.025), :Z=>one(T), :L=>one(T))
        haskey(params, k) || (params[k] = v)
    end
    clr = !isnothing(clearing) ? clearing :
          spec.model === :huggett ? _huggett_clearing(params) :
          _aiyagari_clearing(pfn)
    rb = !isnothing(r_bounds) ? r_bounds :
         spec.model === :huggett ? (T(-0.05), one(T)/spec.individual.beta - one(T) - T(1e-4)) :
         (T(-0.01), T(0.04))
    return _ha_steady_state(spec.individual, spec.grid, spec.income, clr, params;
        K_init=K_init, r_bounds=rb, max_iter=max_iter, tol=tol, verbose=verbose)
end

# Huggett (1993) two-state endowment {e_h, e_l} = {1.0, 0.1},
# transition pi(e_h|e_h)=0.925, pi(e_h|e_l)=0.5  (order [e_h, e_l]).
function _huggett_income()
    states = [1.0, 0.1]
    P = [0.925 0.075; 0.5 0.5]
    # stationary distribution (left eigenvector for eigenvalue 1)
    pi_h = 0.5 / (0.5 + 0.075); stat = [pi_h, 1 - pi_h]
    return IncomeProcess{Float64}(P, states, stat, :endowment)
end

        n_total = n_red + 1
        G1 = zeros(T, n_total, n_total)
        G1[1:n_red, 1:n_red] .= G1_dist
        G1[1:n_red, n_red+1] .= d_channel
        G1[n_red+1, n_red+1] = rho                  # w_{t+1} = rho * w_t
        impact_vec = zeros(T, n_total, 1)
        impact_vec[n_red+1, 1] = one(T)
        impact_vec[1:n_red, 1] .= d_channel
        eigs = eigvals(G1); me = maximum(abs.(eigs))
